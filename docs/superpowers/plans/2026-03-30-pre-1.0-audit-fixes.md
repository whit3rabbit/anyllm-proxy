```rust
/// Interpret backslash escape sequences in double-quoted dotenv values.
/// Only processes \n, \t, \r, \\, and \". All other backslash sequences
/// are passed through unchanged (matches common dotenv behavior).
fn unescape_double_quoted(s: &str) -> String {
    let mut out = String::with_capacity(s.len());
    let mut chars = s.chars();
    while let Some(c) = chars.next() {
        if c == '\\' {
            match chars.next() {
                Some('n') => out.push('\n'),
                Some('t') => out.push('\t'),
                Some('r') => out.push('\r'),
                Some('\\') => out.push('\\'),
                Some('"') => out.push('"'),
                Some(other) => { out.push('\\'); out.push(other); }
                None => out.push('\\'),
            }
        } else {
            out.push(c);
        }
    }
    out
}
```

## Task 6: Validate Bedrock Event Stream CRC32 checksums

**Files:**
- Modify: `crates/proxy/Cargo.toml` (add `crc32fast` dep)

- [ ] **Step 4: Implement CRC validation in EventStreamDecoder::decode_frame**

At the top of `crates/proxy/src/backend/bedrock_client.rs`, add the import:

Find `decode_frame` and the constants at the top of the event stream section. The frame layout is:
- Bytes 0-3: `total_len` (u32 big-endian)
- Bytes 4-7: `headers_len` (u32 big-endian)
- Bytes 8-11: prelude CRC (CRC32 of bytes 0-7)
- Bytes 12 through `total_len-5`: headers + payload
- Bytes `total_len-4` through `total_len-1`: message CRC (CRC32 of bytes 0 through `total_len-5`)

In the `decode_frame` method, after reading `total_len` and `headers_len` but before extracting the payload, add:

```rust
        // Validate prelude CRC (bytes 8-11 = CRC32 of bytes 0-7).
        let prelude_crc_stored = u32::from_be_bytes(self.buf[8..12].try_into().unwrap());
        let prelude_crc_computed = {
            let mut h = Crc32Hasher::new();
            h.update(&self.buf[..8]);
            h.finalize()
        };
        if prelude_crc_stored != prelude_crc_computed {
            let consumed = total_len as usize;
            self.buf.advance(consumed);
            return Err(BedrockClientError::Transport(format!(
                "event stream frame prelude CRC mismatch: stored={prelude_crc_stored:#010x} \
                 computed={prelude_crc_computed:#010x}"
            )));
        }

        // Validate message CRC (last 4 bytes = CRC32 of everything before the message CRC).
        let msg_crc_offset = total_len as usize - 4;
        let msg_crc_stored = u32::from_be_bytes(
            self.buf[msg_crc_offset..msg_crc_offset + 4].try_into().unwrap(),
        );
        let msg_crc_computed = {
            let mut h = Crc32Hasher::new();
            h.update(&self.buf[..msg_crc_offset]);
            h.finalize()
        };
        if msg_crc_stored != msg_crc_computed {
            let consumed = total_len as usize;
            self.buf.advance(consumed);
            return Err(BedrockClientError::Transport(format!(
                "event stream frame message CRC mismatch: stored={msg_crc_stored:#010x} \
                 computed={msg_crc_computed:#010x}"
            )));
        }
```

```rust
    }
```

```rust
fn is_o_series_model(model: &str) -> bool {
    // Matches OpenAI o-series reasoning models: o1, o2, o3, o4, o5, o10, etc.
    // Pattern: starts with 'o' (case-insensitive), followed by one or more ASCII digits,
    // then end-of-string or '-'. Rejects "gpt-4o" (suffix 'o') and "openai-x".
    let bytes = model.as_bytes();
    if bytes.is_empty() || !bytes[0].eq_ignore_ascii_case(&b'o') {
        return false;
    }
    // Must have at least one digit after 'o'.
    let digit_start = 1;
    if bytes.len() <= digit_start || !bytes[digit_start].is_ascii_digit() {
        return false;
    }
    // Remaining digits.
    let after_digits = bytes[digit_start..]
        .iter()
        .position(|b| !b.is_ascii_digit())
        .map(|p| digit_start + p)
        .unwrap_or(bytes.len());
    // After the digits there must be either end-of-string or a '-'.
    after_digits == bytes.len() || bytes[after_digits] == b'-'
}
```
